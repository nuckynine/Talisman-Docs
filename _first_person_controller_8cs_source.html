176 private void OnActionTriggered(InputAction.CallbackContext context)
177 {
178 // Don't go any further if the action doesn't belong to the First Person map.
179 // TODO: Confirm whether we actually need this return.
180 // TODO: If we still need it, work out a better method than a string comparison.
181 if (context.action.actionMap.name != actionMapHandled) return;
182
183 // Determine what we need to do with the action.
184 switch (context.action.name)
185 {
186 case "Look":
187 Look(context);
188 return;
189 case "Move":
190 Move(context);
191 return;
192 case "Crouch":
193 Crouch(context);
194 return;
195 case "Run":
196 Run(context);
197 return;
198 default:
199 // By default, send the input to the current state
200 State newState = state.HandleInput(this, context);
201 if (newState != state.State())
202 {
203 StateTransition(newState);
204 }
205
206 break;
207 }
208 }

275 private void FixedUpdate()
276 {
277 // Handle movement every time step
278 state.HandleMovement(this);
279
280 // Handle viewing every time step
281 CalculateLook(Time.deltaTime);
282
283 // Check to see if the movement has changed our state. If so, change it.
284 State newState = state.CheckForStateTransition(this);
285 if (newState != state.State())
286 {
287 StateTransition(newState);
288 }
289
290 // Actually apply the movement we previously calculated.
291 state.ApplyMovement(this);
292
293
294 }

300 private void StateTransition(State newState)
301 {
302 if (debug) Debug.Log("Transition " + transitionNumber++ + ": from " + state.State() + " to " + newState);
303
304 state.ExitState(this);
305 state = states[(int)newState];
306 state.EnterState(this);
307 }

313 private void CalculateLook(float time)
314 {
315 float horizontal = lookInput.x * time * turnSpeed;
316 float vertical = -lookInput.y * time * turnSpeed;
317
318 yaw += horizontal;
319 pitch += vertical;
320
321 pitch = Mathf.Clamp(pitch, headLowerAngleLimit, headUpperAngleLimit);
322
323 var bodyRotation = Quaternion.AngleAxis(yaw, Vector3.up);
324 var headRotation = Quaternion.AngleAxis(pitch, Vector3.right);
325
326 transform.localRotation = bodyRotation * bodyStartOrientation;
327 head.localRotation = headRotation * headStartOrientation;
328 }
